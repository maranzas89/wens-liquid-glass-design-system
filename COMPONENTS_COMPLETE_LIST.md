## 🚀 使用示例

### Dashboard 示例：
```tsx
import { DashboardPage } from './pages/dashboard-page';

function App() {
  return <DashboardPage />;
}
```

### Login 示例：
```tsx
import { GlassLogin } from './components/glass-login';

function LoginPage() {
  return (
    <div className="min-h-screen flex items-center justify-center">
      <GlassLogin
        onLogin={(email, password) => console.log(email, password)}
        onSignUp={() => navigate('/signup')}
      />
    </div>
  );
}
```

### Charts 示例：
```tsx
import { GlassBarChart, GlassPieChart } from './components/glass-chart';

const data = [
  { label: 'Jan', value: 4000 },
  { label: 'Feb', value: 3000 },
  { label: 'Mar', value: 5000 }
];

<GlassBarChart data={data} height={300} showValues />
<GlassPieChart data={data} size={300} donut />
```
